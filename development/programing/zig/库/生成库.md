# 生成库

## 例子

root.zig

```zig
const std = @import("std");

// 给非 zig 用， 必须有 callconv(.c) i32
export fn add(a: i32, b: i32) callconv(.c) i32 {
    return a + b;
}
```

build.zig

```zig
const lib = b.addLibrary(.{
    // .dynamic 或者 .static
    .linkage = .dynamic,
    .name = "zig_demo",
    .root_module = b.createModule(.{
        .root_source_file = b.path("src/root.zig"),
        .target = target,
        .optimize = optimize,
    }),
});
b.installArtifact(lib);
```
