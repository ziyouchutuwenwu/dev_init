# 调用库

## 分类

### 动态库

main.zig

```zig
const std = @import("std");

const ADD = *const fn (i32, i32) callconv(.c) i32;

pub fn main() !void {
    var lib = try std.DynLib.open("libzig_demo.so");
    defer lib.close();

    const add = lib.lookup(ADD, "add") orelse {
        std.debug.print("Failed to find symbol: add\n", .{});
        return error.SymbolNotFound;
    };

    const result = add(2, 3);
    std.debug.print("Result: {}\n", .{result});
}
```

build.zig

```zig
exe.root_module.linkSystemLibrary("c", .{});
```

### 静态库

main.zig

```zig
const std = @import("std");

extern fn add(a: i32, b: i32) callconv(.c) i32;

pub fn main() !void {
    const result = add(2, 3);
    std.debug.print("Result: {}\n", .{result});
}
```

build.zig

```zig
exe.root_module.addObjectFile(b.path("./libzig_demo.a"));
```
