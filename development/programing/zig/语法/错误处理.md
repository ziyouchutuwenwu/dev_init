# 错误处理

## 用法

### 捕获错误

```zig
const std = @import("std");

pub fn main() void {
    const input = "300";

    const data = std.fmt.parseInt(u8, input, 10);
    const result = data catch |err| {
        switch (err) {
            error.InvalidCharacter => std.debug.print("错误：包含非数字字符\n", .{}),
            error.Overflow => std.debug.print("错误：数值超过 u8 范围\n", .{}),
        }
        return;
    };

    std.debug.print("结果是: {}\n", .{result});
}
```

### 自定义错误

```zig
const std = @import("std");

const MyError = error{
    AAA,
    BBB,
    CCC,
};

fn demo_func(num: i32) !i32 {
    if (num < 0) {
        return MyError.AAA;
    } else {
        return num * num;
    }
}

pub fn main() void {
    const result = demo_func(-111) catch |err| {
        std.log.debug("custom error reason {}", .{err});
        return;
    };
    std.log.debug("result {d}", .{result});
}
```

### errdefer

```zig
const std = @import("std");

pub fn aa() !void {
    errdefer {
        std.log.debug("errdefer 清理垃圾", .{});
    }
    return error.aaa;
}

pub fn main() !void {
    // try aa();

    aa() catch |err| {
        std.log.debug("捕获错误 {}", .{err});
    };

    std.log.debug("正常运行", .{});
}
```
