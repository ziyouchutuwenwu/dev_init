# 交叉编译

## 说明

erlang 交叉编译通过，elixir 就可以用

## 步骤

### erlang

erlang

```bash
https://github.com/erlang/otp
```

toolchain

```bash
buildroot 下，最后的编译器在 build_output/host
复制为 toolchain
```

xcomp.conf

```conf
# target
toolchain=aarch64-buildroot-linux-gnu
erl_xcomp_host=aarch64-buildroot-linux-gnu

# h 文件和 so 寻找的位置
erl_xcomp_isysroot=${TOOLCHAIN_ROOT}/aarch64-buildroot-linux-gnu/sysroot
erl_xcomp_sysroot=${TOOLCHAIN_ROOT}/aarch64-buildroot-linux-gnu/sysroot

# 编译器设置
CC=aarch64-buildroot-linux-gnu-gcc
CPP=aarch64-buildroot-linux-gnu-cpp
CXX=aarch64-buildroot-linux-gnu-g++
LD=aarch64-buildroot-linux-gnu-gcc
RANLIB=aarch64-buildroot-linux-gnu-ranlib
AR=aarch64-buildroot-linux-gnu-ar


# 以下不修改
CFLAGS="-Os -Wall -g -fPIC"
CXXFLAGS="-Os -Wall -fPIC"
LDFLAGS=""
LIBS=
STATIC_CFLAGS="-static"
STATIC_LDFLAGS="-static"

erl_xcomp_build=guess
erl_xcomp_configure_flags="--without-termcap --without-javac"

erl_xcomp_after_morecore_hook=no
erl_xcomp_bigendian=no
erl_xcomp_clock_gettime_cpu_time=yes
erl_xcomp_getaddrinfo=yes
erl_xcomp_gethrvtime_procfs_self=no
erl_xcomp_dlsym_brk_wrappers=no
erl_xcomp_kqueue=no
erl_xcomp_linux_clock_gettime_correction=yes
erl_xcomp_linux_clock_gettime_correction_hz_1000=yes
erl_xcomp_linux_usable_clock_gettime=yes
erl_xcomp_linux_usable_sigev_thread_id=yes
erl_xcomp_poll=yes
erl_xcomp_posix_memalign=yes
erl_xcomp_pwritev=yes
erl_xcomp_readv=no
erl_xcomp_sendfile=yes
erl_xcomp_struct_flock=yes
erl_xcomp_sys_timezone=yes
erl_xcomp_timezone=yes

# pthreads

```

configure

```sh
export BASE_ROOT=$(pwd)
export TOOLCHAIN_ROOT=$BASE_ROOT/toolchain
export PATH=$TOOLCHAIN_ROOT/bin:$PATH

cd otp_src
export ERL_TOP=$(pwd)
# 生成 makefile
./otp_build configure --xcomp-conf=../xcomp.conf
# 先编译一个本机的 erlc, 再用这个 erlc 把代码编译为 beam
./otp_build boot -a
# 打包
./otp_build release -a $BASE_ROOT/beam_erts
```

mix.exs

```elixir
def project do
  [
    .........
    releases: releases()
  ]
end

# 设置 ERTS_PATH，则交叉编译，否则本地编译
defp releases do
  [
    # 注意这里的名字
    demo:
      [steps: [:assemble, :tar]]
      |> then(fn opts ->
        if erts = System.get_env("ERTS_PATH") do
          Keyword.put(opts, :include_erts, erts)
        else
          opts
        end
      end)
  ]
end
```

打包

```bash
cd demo
export ERTS_PATH=$BASE_ROOT/beam_erts/erts-17.0.3
```
