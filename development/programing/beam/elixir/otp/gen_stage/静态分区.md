# 静态分区

## 说明

不同的数据，希望给到不同的消费者去处理

## 例子

```elixir
defmodule Producer do
  use GenStage
  require Logger

  def start_link() do
    GenStage.start_link(__MODULE__, 1, name: __MODULE__)
  end

  def init(counter) do
    Logger.debug("producer init")
    {:producer, counter}
  end

  # consumer 的库存为 min_demand 的时候，就会触发
  # demand 为 max_demand - min_demand
  # 第一次为 max_demand
  def handle_demand(demand, counter) when demand > 0 do
    events = Enum.to_list(counter..(counter + demand - 1))
    Logger.debug("producer demand: #{inspect(demand)}, produce: #{inspect(events)}")
    :timer.sleep(10)
    {:noreply, events, counter + demand}
  end
end
```

```elixir
defmodule ProducerConsumer2 do
  use GenStage
  require Logger

  def start_link() do
    GenStage.start_link(__MODULE__, :ok, name: __MODULE__)
  end

  def init(:ok) do
    Logger.debug("2 init")

    # 静态分区路由
    dispatcher =
      {GenStage.PartitionDispatcher,
       partitions: [:small, :large],
       hash: fn n ->
         if n < 50 do
           {n, :small}
         else
           {n, :large}
         end
       end}

    {:producer_consumer, :ok, dispatcher: dispatcher}
  end

  def handle_events(events, _from, state) do
    data_to_send = Enum.map(events, fn n -> n + 1 end)
    Logger.debug("2 recv: #{inspect(events)}, send: #{inspect(data_to_send)}")
    :timer.sleep(20)
    {:noreply, data_to_send, state}
  end
end
```

```elixir
defmodule ConsumerLarge do
  use GenStage
  require Logger

  def start_link() do
    GenStage.start_link(__MODULE__, :ok, name: __MODULE__)
  end

  def init(:ok) do
    Logger.debug("large init")
    {:consumer, :ok}
  end

  def handle_events(events, _from, state) do
    Enum.each(events, fn event ->
      Logger.debug("large: #{event}")
      :timer.sleep(200)
    end)
    {:noreply, [], state}
  end
end
```

```elixir
defmodule Demo do
  def demo do
    Producer.start_link()
    ProducerConsumer2.start_link()
    ConsumerSmall.start_link()
    ConsumerLarge.start_link()

    GenStage.sync_subscribe(
      ProducerConsumer2,
      to: Producer,
      max_demand: 22,
      min_demand: 11
    )

    GenStage.sync_subscribe(
      ConsumerSmall,
      to: ProducerConsumer2,
      max_demand: 33,
      min_demand: 22,
      partition: :small
    )

    GenStage.sync_subscribe(
      ConsumerLarge,
      to: ProducerConsumer2,
      max_demand: 33,
      min_demand: 22,
      partition: :large
    )
  end
end
```
