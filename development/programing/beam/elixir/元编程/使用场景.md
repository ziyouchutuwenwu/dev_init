## 场景

### 动态添加方法和属性

use 模块除了在模块调用 `__using__` 宏外, 不做其他任何事情

注意 `require` 的位置

```elixir
defmodule MacroDemo do
  defmacro demo(name) do
    quote do
      unquote(name) <> ", " <> "in SimpleMacro"
    end
  end
end

defmodule UsingDemo do
  defmacro __using__(opts) do
    params = Keyword.get(opts, :aaa, "default params")

    quote do
      require MacroDemo
      def demo(name) do
        value = "value from " <> unquote(params) <> ", " <> name
        MacroDemo.demo(value)
      end
    end
  end
end

defmodule MyModule do
  use UsingDemo, aaa: "123"
end
```

测试

```elixir
MyModule.demo("me")
```

### 类似继承，允许其他模块重写

- 方法重写

```elixir
defmodule DefaultMod do
  defmacro __using__(_opts) do

    # 动态添加的方法
    quote do
      def test1(x, y) do
        x + y
      end

      def test2(x, y) do
        x - y
      end
      defoverridable test1: 2, test2: 2
    end
  end
end

defmodule InheritMod do
  use DefaultMod

  def test1(x, y) do
    x * y
  end

  def test2(x, y) do
    x * y * y
  end
end
```

测试

```elixir
InheritMod.test1(11, 22)
InheritMod.test2(11, 22)
```

- Behaviour 重写

```elixir
# 定义 Behaviour
defmodule MyBehaviour do
  @callback test(number(), number()) :: number()
end

# defoverridable 使用 Behaviour 当参数的时候，所有 callback 都可以被 override
defmodule DefaultMod do
  defmacro __using__(_opts) do

    # 动态添加的方法
    quote do
      @behaviour MyBehaviour
      def test(x, y) do
        "default"
        x + y
      end

      defoverridable MyBehaviour
    end
  end
end

defmodule InheritMod do
  use DefaultMod

  def test(x, y) do
    "override"
    x * y
  end
end
```

测试

```elixir
InheritMod.test(11, 22)
```

### 动态生成常量

```elixir
defmodule Foo do
  defmacro add(name) do
    quote do
      @xxx unquote(name)
    end
  end
end

defmodule Bar do
  require Foo
  import Foo

  Module.register_attribute(__MODULE__, :xxx, accumulate: true)

  add "john"
  add "henry"

  IO.inspect(@xxx)
end
```

### 类似 java 的注解

```elixir
defmodule DemoMacro do
  defmacro __using__(_) do
    quote do
      Module.register_attribute(__MODULE__, :xxx, accumulate: true)
    end
  end

  defmacro show(name) do
    quote do
      IO.inspect("#{unquote(name)} #{@xxx}")
    end
  end
end

defmodule Demo do
  use DemoMacro

  @xxx "测试"

  def aaa() do
    DemoMacro.show("123")
  end
end
```

### 编译期 hook

```elixir
defmodule Foo do
  defmacro __using__(_) do
    quote do
      @before_compile unquote(__MODULE__)
    end
  end

  defmacro __before_compile__(env) do
    IO.inspect(env)
    nil
  end
end

defmodule Bar do
  use Foo
end
```

### 模拟 test 的实现

```elixir
defmodule Calculator do
  def add(a, b), do: a + b
  def subtract(a, b), do: a - b
end

defmodule TestCase do
  defmacro __using__(_) do
    quote do
      import TestCase
      Module.register_attribute(__MODULE__, :tests, accumulate: true)
      @before_compile unquote(__MODULE__)
    end
  end

  defmacro __before_compile__(_env) do
    # Inject a run function into the test case after all tests have been accumulated
    quote do
      def run do
        Enum.each @tests, fn test_name ->
          result = apply(__MODULE__, test_name, [])
          state = if result, do: "pass", else: "fail"
          IO.puts "#{test_name} => #{state}"
        end
      end
    end
  end

  defmacro test(description, do: body) do
    test_name = String.to_atom(description)
    quote do
      @tests unquote(test_name)
      def unquote(test_name)(), do: unquote(body)
    end
  end
end

defmodule CalculatorTest do
  use TestCase
  import Calculator

  test "add 1, 2 should return 3" do
    add(1, 2) == 3
  end

  test "subtract 5, 2 should not return 4" do
    subtract(5, 2) == 4
  end
end

CalculatorTest.run
```
