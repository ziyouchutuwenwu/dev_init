# 自动注册

## 说明

设备通过 JITP 的模式，自动注册到后端

关于证书

```sh
ca
├── device
├── rootca
├── verify
└── website
```

```sh
rootca  所有证书的基础，自己颁发
verify  验证 rootca，表明 rootca 是自己的
website 设备通过 ssl 和 web 通信用的
device  给设备签发的证书
```

## 配置

### rootca

生成

```sh
# 生成 root ca 私钥
openssl genrsa -out root-ca-key.pem 4096

# 生成自签名 root ca 证书
# /CN 是通用名
# /O 是组织
# /C 是国家
openssl req -x509 -new -nodes -key root-ca-key.pem \
  -sha256 -days 7300 \
  -subj "/CN=测试的复现 CA 证书/O=测试组织/C=CN" \
  -out root-ca.pem

# 验证，确认输出包含 CA:TRUE
openssl x509 -in root-ca.pem -text -noout | rg TRUE
```

### 网站证书

```sh
# 生成服务端私钥
openssl genrsa -out server-key.pem 2048

# 生成自签名服务端证书
openssl req -x509 -new -nodes -key server-key.pem \
  -sha256 -days 365 \
  -subj "/CN=测试的网站证书/O=测试网站/C=CN" \
  -out server-cert.pem
```

### 设备证书

每个设备做一个，根据设备 id 来

```sh
# 设备私钥
openssl genrsa -out key.pem 2048

# 每个设备有一个 id
# 比如 xxx-device-001
openssl req -new -key key.pem \
  -subj "/CN=xxx-device-001/O=NervesHub/C=CN" \
  -out device.csr

# 用 root ca 签名
openssl x509 -req -in device.csr \
  -CA ../rootca/root-ca.pem -CAkey ../rootca/root-ca-key.pem \
  -CAcreateserial -days 365 -sha256 \
  -out cert.pem

# 验证
openssl verify -CAfile ../rootca/root-ca.pem cert.pem
# → device-cert.pem: OK

rm -rf device.csr
```

### 设备

deps

```elixir
{:nerves_hub_link, "~> 2.2"},
```

configurator.ex

```elixir
defmodule DeviceCert.Configurator do
  @behaviour NervesHubLink.Configurator

  @impl NervesHubLink.Configurator
  def build(config) do
    ssl =
      config.ssl
      |> Keyword.put(:certfile, "/data/cert/cert.pem")
      |> Keyword.put(:keyfile, "/data/cert/key.pem")
      |> Keyword.put(:verify, :verify_none)
      |> Keyword.put(:versions, [:"tlsv1.2"])

    %{config | ssl: ssl}
  end
end
```

config/target.exs

```elixir
config :nerves_hub_link,
  host: "10.0.2.1:4001",
  remote_iex: true,
  connect_wait_for_network: false,
  configurator: DeviceCert.Configurator,
  socket: [
    path: "/socket/websocket",
    tls_opts: [
      versions: [:"tlsv1.2"]
    ]
  ]
```
