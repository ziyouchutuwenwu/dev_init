```elixir
defmodule WebDemoWeb.Router do
  use WebDemoWeb, :router

  pipeline :browser do
    plug :accepts, ["html"]
    plug :fetch_session
    plug :fetch_live_flash
    plug :put_root_layout, html: {WebDemoWeb.Layouts, :root}
    plug :protect_from_forgery
    plug :put_secure_browser_headers
  end

  pipeline :api do
    plug :accepts, ["json"]
  end

  scope "/", WebDemoWeb do
    pipe_through :browser

    get "/", PageController, :home
  end

  scope "/file", WebDemoWeb do
    pipe_through :api

    get "/download", FileController, :download
    put "/put-upload", FileController, :put_upload
    post "/read-upload", FileController, :read_upload
    put "/stream-upload", FileController, :stream_upload
  end

  scope "/api", WebDemoWeb do
    pipe_through :api

    get "/get", DemoController, :get
    get "/uri/:uri_arg", DemoController, :uri_get
    post "/form-post", DemoController, :form_post
    post "/json-post", DemoController, :json_post
  end
end
```

### controller

demo_controller.ex

```elixir
defmodule WebDemoWeb.DemoController do
  require Logger

  use WebDemoWeb, :controller

  def get(conn, params) do
    %{"aaa" => value} = params
    text(conn, "value is #{value}")
  end

  def uri_get(conn, params) do
    %{"uri_arg" => uri_arg_value} = params
    text(conn, "uri #{uri_arg_value}")
  end

  def form_post(conn, params) do
    text(conn, inspect(params))
  end

  def json_post(conn, _params) do
    json = Jason.encode!(conn.body_params)
    text(conn, json)
  end
end
```

file_controller.ex

```elixir
defmodule WebDemoWeb.FileController do
  require Logger
  use WebDemoWeb, :controller

  def read_upload(conn, params) do
    if upload = params["file"] do
      filename = upload.filename
      Logger.debug("filename #{inspect(filename)}")
      File.cp(upload.path, "/tmp/saved_read_bin")
    end

    json(conn, %{msg: "上传成功"})
  end

  def put_upload(conn, _params) do
    # {:ok, file_bin, _conn} = Plug.Conn.read_body(conn)
    {:ok, file_bin, _conn} =
      Plug.Conn.read_body(conn, length: 100_000_000, read_length: 2_000_000, read_timeout: 30_000)

    result = File.write("/tmp/saved_put_bin", file_bin)
    json(conn, %{msg: inspect(result)})
  end

  def stream_upload(conn, _params) do
    {:ok, file} = File.open("/tmp/saved_stream_bin", [:write, :binary])

    try do
      case loop_read_stream_write(conn, file) do
        {:ok, conn} ->
          Logger.info("文件上传成功")
          send_resp(conn, 200, "上传成功")

        {:error, _reason, conn} ->
          Logger.error("文件上传过程中发生错误")
          send_resp(conn, 500, "Internal Server Error")
      end
    after
      File.close(file)
    end
  end

  defp loop_read_stream_write(conn, file) do
    case Plug.Conn.read_body(conn, length: 20_000_000) do
      {:ok, data, conn} ->
        IO.binwrite(file, data)
        {:ok, conn}

      {:more, data, conn} ->
        IO.binwrite(file, data)
        loop_read_stream_write(conn, file)

      {:error, reason} ->
        {:error, reason, conn}
    end
  end

  def download(conn, _params) do
    file_path = System.get_env("HOME") |> Path.join("downloads") |> Path.join("todo")
    conn |> send_download({:file, file_path})
  end
end
```

### 大小限制

上传文件大小限制

endpoint.ex

```elixir
plug Plug.Parsers,
  parsers: [:urlencoded, :multipart, :json],
```

改成

```elixir
plug Plug.Parsers,
  # multipart 请求大小限制为 20M
  parsers: [
    # x-www-form-urlencoded
    :urlencoded,
    {
      # multipart/form-data
      :multipart,
      # length 是总大小
      # read_length 单次读取大小
      # read_timeout 单次读取超时
      length: 20_000_000_000, read_length: 2_000_000, read_timeout: 30_000
    },
    :json
  ],
```
