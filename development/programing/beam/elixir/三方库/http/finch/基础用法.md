## 例子

### 项目

```sh
mix new client_demo
```

### deps

```elixir
{:jason, "~> 1.2"},
```

### 配置

config/config.exs

```elixir
import Config

config :client_demo, mode: config_env()
import_config "#{config_env()}.exs"
```

config/dev.exs

```elixir
import Config

config :logger,
  level: :debug

config :client_demo,
  base_url: "http://127.0.0.1:4000"
```

config/prod.exs

```elixir
import Config

config :logger,
  level: :warning

config :client_demo,
  base_url: "http://127.0.0.1:4000"
```

config/test.exs

```elixir
import Config

config :logger,
  level: :debug

config :client_demo,
  base_url: "http://127.0.0.1:4000"
```

config/runtime.exs

```elixir
import Config

if config_env() == :prod do
  if System.get_env("BASE_URL") do
    config :client_demo,
      base_url: System.get_env("BASE_URL")
  end
end
```

config_fetcher.ex

```elixir
defmodule ConfigFetcher do
  def get_base_url() do
    Application.get_env(:client_demo, :base_url)
  end
end
```

### 代码

```elixir
defmodule DemoApp do
  use Application
  require Logger

  def start(_type, _args) do
    Logger.debug("in app start")

    children = [
      {
        Finch,
        name: ConfigedFinch,
        pools: %{
          default: [size: 98]
        }
      }
    ]

    Supervisor.start_link(children, strategy: :one_for_one)
  end
end
```

```elixir
defmodule ClientDemo do
  require Logger

  def get do
    headers = [
      {"User-Agent", "xxx"}
    ]

    base_url = ConfigFetcher.get_base_url() |> URI.parse()

    url =
      base_url
      |> URI.append_path("/api/get")
      |> URI.append_query("aaa=bbb")
      |> URI.to_string()

    {:ok, resp} = Finch.build(:get, url, headers) |> Finch.request(ConfigedFinch)

    body = resp.body

    Logger.debug(body)
  end

  def uri_get do
    headers = [
      {"User-Agent", "xxx"}
    ]

    base_url = ConfigFetcher.get_base_url() |> URI.parse()

    url =
      base_url
      |> URI.append_path("/api/uri/ppp")
      |> URI.to_string()

    {:ok, resp} = Finch.build(:get, url, headers) |> Finch.request(ConfigedFinch)

    body = resp.body

    Logger.debug(body)
  end

  def form_post do
    headers = [
      {"User-Agent", "xxx"},
      {"Content-Type", "application/x-www-form-urlencoded"}
    ]

    form_data = URI.encode_query(name: "name1", age: "age1")

    base_url = ConfigFetcher.get_base_url() |> URI.parse()

    url =
      base_url
      |> URI.append_path("/api/form-post")
      |> URI.to_string()

    {:ok, resp} = Finch.build(:post, url, headers, form_data) |> Finch.request(ConfigedFinch)

    body = resp.body
    Logger.debug(body)
  end

  def json_post do
    headers = [
      {"User-Agent", "xxx"},
      {"Content-Type", "application/json"}
    ]

    json_data = %{
      aaa: "bbb"
    }

    base_url = ConfigFetcher.get_base_url() |> URI.parse()

    url =
      base_url
      |> URI.append_path("/api/json-post")
      |> URI.to_string()

    body = Jason.encode!(json_data)

    {:ok, resp} = Finch.build(:post, url, headers, body) |> Finch.request(ConfigedFinch)

    body = resp.body
    Logger.debug(body)
  end
end
```

## 类比 curl

```sh
curl 'http://127.0.0.1:4000/api/get-demo?aaa=bbb'
curl 'http://127.0.0.1:4000/api/uri-demo/ppp'
curl 'http://127.0.0.1:4000/api/form-post' -d 'abc=xyz'
curl 'http://127.0.0.1:4000/api/json-post' -H "Content-type: application/json" -d '{"phone": "18000011005","password": "xxxxx"}'
```
