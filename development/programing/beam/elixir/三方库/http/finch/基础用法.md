### 配置

config/config.exs

```elixir
import Config

config :client_demo, mode: config_env()
import_config "#{config_env()}.exs"
```

config/dev.exs

```elixir
import Config

config :logger,
  level: :debug

config :client_demo,
  base_url: "http://127.0.0.1:4000"

config :client_demo,
  finch_options: [
    # 发送请求到收到数据的总时间
    # sse 为从发请求到所有 chunk 都收到以后的时间
    request_timeout: 1000,
    # 收到数据的时间
    # sse 为收到每个 chunk 的时间
    receive_timeout: 1000,

    # pool 里面取出来的时间，和请求没啥关系
    pool_timeout: 1000
  ]
```

```elixir

if config_env() == :prod do
  if System.get_env("BASE_URL") do
    config :client_demo,
      base_url: System.get_env("BASE_URL")
  end
end
```

config_fetcher.ex

```elixir
defmodule ConfigFetcher do
  def get_base_url() do
    Application.get_env(:client_demo, :base_url)
  end

  def get_finch_options() do
    Application.get_env(:client_demo, :finch_options)
  end
end
```

### 代码

先做全局配置

```elixir
defmodule FinchWrapper do
  def do_request(req, options) do
    default_headers = %{
      "User-Agent" => "xxx",
      "aaa" => "bbb"
    }

    request_headers = Map.new(req.headers)
    final_headers = Map.merge(default_headers, request_headers) |> Map.to_list()

    req = %{req | headers: final_headers}
    Finch.request(req, ConfigedFinch, options)
  end
end
```

```elixir
defmodule ClientDemo do
  require Logger

  def get do
    base_url = ConfigFetcher.get_base_url() |> URI.parse()

    url =
      base_url
      |> URI.append_path("/api/get")
      |> URI.append_query("aaa=bbb")
      |> URI.to_string()

    options = ConfigFetcher.get_finch_options()

    # {:ok, resp} = Finch.build(:get, url, headers) |> Finch.request(ConfigedFinch, options)
    {:ok, resp} = Finch.build(:get, url) |> FinchWrapper.do_request(options)

    body = resp.body

    Logger.debug(body)
  end

  def uri_get do
    base_url = ConfigFetcher.get_base_url() |> URI.parse()

    url =
      base_url
      |> URI.append_path("/api/uri/ppp")
      |> URI.to_string()

    options = ConfigFetcher.get_finch_options()
    {:ok, resp} = Finch.build(:get, url) |> FinchWrapper.do_request(options)

    body = resp.body

    Logger.debug(body)
  end

  def form_post do
    form_data = URI.encode_query(name: "name1", age: "age1")

    base_url = ConfigFetcher.get_base_url() |> URI.parse()

    headers = [
      {"Content-Type", "application/x-www-form-urlencoded"}
    ]

    url =
      base_url
      |> URI.append_path("/api/form-post")
      |> URI.to_string()

    options = ConfigFetcher.get_finch_options()

    {:ok, resp} =
      Finch.build(:post, url, headers, form_data) |> FinchWrapper.do_request(options)

    body = resp.body
    Logger.debug(body)
  end

  def json_post do
    json_data = %{
      aaa: "bbb"
    }

    base_url = ConfigFetcher.get_base_url() |> URI.parse()

    headers = [
      {"Content-Type", "application/json"}
    ]

    url =
      base_url
      |> URI.append_path("/api/json-post")
      |> URI.to_string()

    body = Jason.encode!(json_data)

    options = ConfigFetcher.get_finch_options()
    {:ok, resp} = Finch.build(:post, url, headers, body) |> FinchWrapper.do_request(options)

    body = resp.body
    Logger.debug(body)
  end
end
```
