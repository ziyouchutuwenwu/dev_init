# 文件上传

## 例子

### 创建

```sh
mix new file_demo --sup
```

### 代码

先做全局配置

```elixir
defmodule FileDemo do
  require Logger

  def read_upload do
    file_path = System.get_env("HOME") |> Path.join("downloads") |> Path.join("aaaaa")
    file_bin = File.read!(file_path)
    filename = Path.basename(file_path)

    boundary = "----------#{System.unique_integer([:positive])}"

    body = [
      "\r\n--#{boundary}\r\n",
      "Content-Disposition: form-data; name=\"file\"; filename=\"#{filename}\"\r\n",
      "Content-Type: application/octet-stream\r\n\r\n",
      file_bin,
      "\r\n--#{boundary}--\r\n"
    ]

    headers = [
      {"User-Agent", "xxx"},
      {"Content-Type", "multipart/form-data; boundary=#{boundary}"}
    ]

    base_url = ConfigFetcher.get_base_url() |> URI.parse()
    url = base_url |> URI.append_path("/file/read-upload") |> URI.to_string()

    options = ConfigFetcher.get_finch_options()
    {:ok, resp} = Finch.build(:post, url, headers, body) |> Finch.request(ConfigedFinch, options)

    resp_body = resp.body
    Logger.debug(resp_body)
  end

  def put_upload do
    file_path = System.get_env("HOME") |> Path.join("downloads") |> Path.join("aaaaa")
    file_bin = File.read!(file_path)

    headers = [
      {"User-Agent", "xxx"},
      {"Content-Type", "application/octet-stream"}
    ]

    base_url = ConfigFetcher.get_base_url() |> URI.parse()
    url = base_url |> URI.append_path("/file/put-upload") |> URI.to_string()

    options = ConfigFetcher.get_finch_options()

    {:ok, resp} =
      Finch.build(:put, url, headers, file_bin) |> Finch.request(ConfigedFinch, options)

    body = resp.body
    Logger.debug(body)
  end

  def stream_upload do
    file_path = System.get_env("HOME") |> Path.join("downloads") |> Path.join("aaaaa")
    filename = Path.basename(file_path)
    boundary = "----------#{System.unique_integer([:positive])}"

    header_chunks = [
      "\r\n--#{boundary}\r\n",
      "Content-Disposition: form-data; name=\"file\"; filename=\"#{filename}\"\r\n",
      "Content-Type: application/octet-stream\r\n\r\n"
    ]

    footer_chunk = "\r\n--#{boundary}--\r\n"

    content_length =
      IO.iodata_length(header_chunks) + File.stat!(file_path).size + byte_size(footer_chunk)

    body_stream =
      Stream.concat([header_chunks, File.stream!(file_path, 2_000_000), [footer_chunk]])

    headers = [
      {"User-Agent", "xxx"},
      {"Content-Type", "multipart/form-data; boundary=#{boundary}"},
      {"Content-Length", Integer.to_string(content_length)}
    ]

    base_url = ConfigFetcher.get_base_url() |> URI.parse()
    url = base_url |> URI.append_path("/file/stream-upload") |> URI.to_string()
    options = ConfigFetcher.get_finch_options()

    {:ok, resp} =
      Finch.build(:post, url, headers, {:stream, body_stream})
      |> Finch.request(
        ConfigedFinch,
        Keyword.merge(options || [],
          request_timeout: :infinity,
          receive_timeout: :infinity
        )
      )

    Logger.debug(resp.body)
  end
end
```
