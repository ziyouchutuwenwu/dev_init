# 文件上传

## 例子

```elixir
defmodule FileDemo do
  require Logger

  def read_upload do
    file_path = System.get_env("HOME") |> Path.join("downloads") |> Path.join("aaaaa")
    file_bin = File.read!(file_path)
    filename = Path.basename(file_path)

    boundary = "----------#{System.unique_integer([:positive])}"

    body = [
      "\r\n--#{boundary}\r\n",
      "Content-Disposition: form-data; name=\"file\"; filename=\"#{filename}\"\r\n",
      "Content-Type: application/octet-stream\r\n\r\n",
      file_bin,
      "\r\n--#{boundary}--\r\n"
    ]

    headers = [
      {"Content-Type", "multipart/form-data; boundary=#{boundary}"}
    ]

    base_url = ConfigFetcher.get_base_url() |> URI.parse()
    url = base_url |> URI.append_path("/file/read-upload") |> URI.to_string()

    options = ConfigFetcher.get_finch_options()
    {:ok, resp} = Finch.build(:post, url, headers, body) |> FinchWrapper.do_request(options)

    resp_body = resp.body
    Logger.debug(resp_body)
  end

  def put_upload do
    file_path = System.get_env("HOME") |> Path.join("downloads") |> Path.join("aaaaa")
    file_bin = File.read!(file_path)

    headers = [
      {"Content-Type", "application/octet-stream"}
    ]

    base_url = ConfigFetcher.get_base_url() |> URI.parse()
    url = base_url |> URI.append_path("/file/put-upload") |> URI.to_string()

    options = ConfigFetcher.get_finch_options()

    {:ok, resp} =
      Finch.build(:put, url, headers, file_bin) |> FinchWrapper.do_request(options)

    body = resp.body
    Logger.debug(body)
  end

  def stream_upload do
    file_path = System.get_env("HOME") |> Path.join("downloads/aaaaa")
    file_size = File.stat!(file_path).size
    body_stream = File.stream!(file_path, 1_000_000)

    headers = [
      {"Content-Type", "application/octet-stream"},
      {"Content-Length", Integer.to_string(file_size)}
    ]

    base_url = ConfigFetcher.get_base_url() |> URI.parse()
    url = base_url |> URI.append_path("/file/stream-upload") |> URI.to_string()
    options = ConfigFetcher.get_finch_options()

    {:ok, resp} =
      Finch.build(:put, url, headers, {:stream, body_stream}) |> FinchWrapper.do_request(options)

    Logger.debug(resp.body)
  end
end
```
