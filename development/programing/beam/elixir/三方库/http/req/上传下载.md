# 文件上传

## 例子

```elixir
defmodule FileDemo do
  require Logger

  def read_upload() do
    file_path = System.get_env("HOME") |> Path.join("downloads/aaaaa")
    file_bin = file_path |> File.read!()

    base_url = ConfigFetcher.get_base_url() |> URI.parse()
    url = base_url |> URI.append_path("/file/read-upload") |> URI.to_string()

    client = ReqWrapper.build()

    client
    |> Req.post!(
      url: url,
      form_multipart: [
        file: {file_bin, filename: Path.basename(file_path)}
      ]
    )
  end

  def put_upload do
    file_path = System.get_env("HOME") |> Path.join("downloads/aaaaa")
    file_bin = file_path |> File.read!()

    base_url = ConfigFetcher.get_base_url() |> URI.parse()
    url = base_url |> URI.append_path("/file/put-upload") |> URI.to_string()

    client = ReqWrapper.build()
    {:ok, resp} = client |> Req.put(url: url, body: file_bin)
    body = resp.body

    Logger.debug(body)
  end

  def stream_upload do
    file_path = System.get_env("HOME") |> Path.join("downloads/aaaaa")
    file_size = File.stat!(file_path).size
    body_stream = File.stream!(file_path, 1_000_000)

    base_url = ConfigFetcher.get_base_url() |> URI.parse()
    url = base_url |> URI.append_path("/file/stream-upload") |> URI.to_string()

    client = ReqWrapper.build()

    request =
      client
      |> Req.merge(
        url: url,
        body: body_stream,
        headers: [
          {"Content-Type", "application/octet-stream"},
          {"Content-Length", Integer.to_string(file_size)}
        ]
      )

    resp = Req.put!(request)
    Logger.debug(resp.body)
  end
end
```
