## 例子

```elixir
defmodule ReqWrapper do
  def build() do
    connect_opts =
      case System.get_env("HTTP_PROXY") do
        nil ->
          []

        proxy_url ->
          proxy = URI.parse(proxy_url)
          [proxy: {:http, proxy.host, proxy.port, []}]
      end

    Req.new(
      connect_options: connect_opts,
      headers: [
        {"User-Agent", "xxx"}
      ],
      retry: false,
      receive_timeout: :infinity
      # 使用全局 finch 配置
      # finch: ConfigedFinch
    )
  end
end
```
