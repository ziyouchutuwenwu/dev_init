        margin-top: 24px;
        font-size: 13px;
        color: #94a3b8;
      }
      a {
        color: #4f46e5;
      }
    </style>
  </head>
  <body>
    <div class="card">
      <h1>Dashboard</h1>
      <div id="status">Checking...</div>
      <div class="info">You can close this page.</div>
    </div>
    <script>
      (function () {
        var status = document.getElementById("status");
        function getCookie(name) {
          var m = document.cookie.match(new RegExp("(?:^|; )" + name + "=([^;]*)"));
          return m ? decodeURIComponent(m[1]) : null;
        }
        var token = getCookie("session_token");
        if (token) {
          try {
            var decoded = atob(token);
            var parts = decoded.split(":");
            status.innerText = "Logged in as: " + parts[0];
            status.className = "auth-ok";
          } catch (_) {
            status.innerText = "Invalid session token";
            status.className = "auth-fail";
          }
        } else {
          status.innerText = "Not authenticated!";
          status.className = "auth-fail";
        }
      })();
    </script>
  </body>
</html>
