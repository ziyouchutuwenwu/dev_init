"use strict";

const _s = {
  icon: (s) => (s === "safe" ? "✅" : s === "warning" ? "⚠️" : s === "pending" ? "⏳" : "❌"),
  cls: (s) => s,
  badge: (s) => ({ safe: "安全", warning: "可疑", danger: "风险", pending: "检测中" })[s] || s,
};

function calcScore(results) {
  let earned = 0,
    maxScore = 0;
  for (const r of results) {
    const w = r.weight || 1;
    maxScore += w;
    if (r.status === "safe") earned += w;
    else if (r.status === "warning") earned += w * 0.5;
  }
  return { earned, maxScore, pct: Math.round((earned / maxScore) * 100) };
}

function updateHeader(results, extras) {
  const total = results.length;
  const danger = results.filter((r) => r.status === "danger").length;
  const warning = results.filter((r) => r.status === "warning").length;
  const safe = results.filter((r) => r.status === "safe").length;
  const { pct } = calcScore(results);

  document.getElementById("total-count").textContent = total;

  const bar = document.getElementById("score-bar");
  bar.style.width = pct + "%";
  bar.className = pct >= 70 ? "bar-safe" : pct >= 40 ? "bar-warning" : "bar-danger";

  let label = "信任度 " + pct + "%（" + safe + " 安全 · " + warning + " 可疑 · " + danger + " 风险）";
  if (extras) {
    label += " | " + (Array.isArray(extras) ? extras.join(" · ") : extras);
  }
  document.getElementById("score-label").textContent = label;
}

function renderResults(results) {
  const groups = {};
  for (const r of results) {
    const key = r.label || "其他";
    if (!groups[key]) groups[key] = [];
    groups[key].push(r);
  }

  const main = document.getElementById("results");
  main.innerHTML = "";

  for (const [label, items] of Object.entries(groups)) {
    const section = document.createElement("section");
    section.className = "group";
    const h2 = document.createElement("h2");
    h2.textContent = label + "（" + items.length + "）";
    section.appendChild(h2);
    const list = document.createElement("ul");
    list.className = "check-list";

    for (const item of items) {
      const li = document.createElement("li");
      li.className = "check-item " + _s.cls(item.status);

      const iconSpan = document.createElement("span");
      iconSpan.className = "icon";
      iconSpan.textContent = _s.icon(item.status);
      li.appendChild(iconSpan);

      const nameSpan = document.createElement("span");
      nameSpan.className = "name";
      nameSpan.textContent = item.name;
      li.appendChild(nameSpan);

      const statusSpan = document.createElement("span");
      statusSpan.className = "badge";
      statusSpan.textContent = _s.badge(item.status);
      li.appendChild(statusSpan);

      const detailSpan = document.createElement("span");
      detailSpan.className = "detail";
      detailSpan.textContent = item.detail || "";
      li.appendChild(detailSpan);

      const weightSpan = document.createElement("span");
      weightSpan.className = "weight";
      weightSpan.textContent = "×" + (item.weight || 1);
      li.appendChild(weightSpan);

      list.appendChild(li);
    }
    section.appendChild(list);
    main.appendChild(section);
  }
}

document.addEventListener("DOMContentLoaded", async () => {

"use strict";

class CDPDetector {
  constructor() {
    this.results = [];
    this.totalWeight = 0;
    this.score = 100;
  }

  get _groupAll() {
    return "检测聚合组";
  }

  async run() {
    this.results = [];
    this.totalWeight = 0;

    const checks = [
      ...this.groupWebdriver(),
      ...this.groupChrome(),
      ...this.groupConsole(),
      ...this.groupPlugin(),
      ...this.groupHardware(),
      ...this.groupScreen(),
      ...this.groupUA(),
      ...this.groupWebGL(),
      ...this.groupMedia(),
      ...this.groupFunction(),
      ...this.groupTime(),
      ...this.groupAPI(),
      ...this.groupAudio(),
      ...this.groupWebRTC(),
      ...this.groupCanvasFont(),
      ...this.groupClientRects(),
      ...this.groupAntiTamper(),
      ...this.groupNetwork(),
      ...this.groupWorker(),
      ...this.groupSensors(),
      ...this.groupHardwareAPI(),
      ...this.groupObservers(),
      ...this.groupStorage(),
      ...this.groupMiscAPI(),
      ...this.groupMisc(),
      ...this.groupIntegrity(),
    ];

    const asyncChecks = [];
    for (const c of checks) {
      this.totalWeight += c.weight || 1;
      const ret = c.fn();
      if (ret && typeof ret.then === "function") {
        asyncChecks.push(ret.then((r) => this.results.push({ ...c, ...r })));
      } else {
        this.results.push({ ...c, ...ret });
      }
    }
    await Promise.all(asyncChecks);

    let earned = 0;
    for (const r of this.results) {
      const w = r.weight || 1;
      if (r.status === "safe") earned += w;
      else if (r.status === "warning") earned += w * 0.5;
    }
    this.score = Math.round((earned / this.totalWeight) * 100);

    return { score: this.score, results: Object.freeze(this.results.slice()) };
  }

  groupWebdriver() {
    const label = "自动化标记";
    return [
      {
        name: "navigator.webdriver 是否为 undefined",
        label,
        weight: 3,
        fn: () => {
          const wd = Object.getOwnPropertyDescriptor(navigator, "webdriver");
          if (wd === undefined) return { status: "safe", detail: "webdriver 属性已被完全移除（理想状态）" };
          return { status: "danger", detail: "webdriver 属性仍然存在，可能被设为 " + navigator.webdriver };
        },
      },
      {
        name: "navigator.webdriver 值",
        label,
        weight: 2,
        fn: () => {
          const v = navigator.webdriver;
          if (v === undefined) return { status: "safe", detail: "属性已移除" };
          if (v === true) return { status: "danger", detail: "webdriver === true，明确表示受自动化控制" };
          if (v === false) return { status: "warning", detail: "webdriver === false，此地无银，说明被篡改过" };
          return { status: "safe", detail: "值为 " + JSON.stringify(v) };
        },
      },
      {
        name: "webdriver getter 是否原生",
        label,
        weight: 2,
        fn: () => {
          const desc = Object.getOwnPropertyDescriptor(navigator, "webdriver");
          if (desc === undefined) return { status: "safe", detail: "属性不存在，无法检测 getter" };
          const getterStr = (desc.get || "").toString();
          if (getterStr.includes("native code")) {
            return { status: "warning", detail: "getter 虽为原生函数，但属性本身被暴露" };
          }
          return { status: "danger", detail: "getter 已被篡改: " + getterStr.slice(0, 80) };
        },
      },
      {
        name: "navigator.plugins 是否可枚举",
        label,
        weight: 1,
        fn: () => {
          try {
            for (const k in navigator) {
              if (k === "plugins" || k === "webdriver") {
                return { status: "danger", detail: "navigator." + k + " 可被 for-in 枚举，异常泄露" };
              }
            }
          } catch (_) {}
          return { status: "safe", detail: "navigator 属性未异常泄露" };
        },
      },
    ];
  }

  groupChrome() {
    const label = "Chrome 对象特征";
    return [
      {
        name: "window.chrome 是否存在",
        label,
        weight: 3,
        fn: () => {
          if (!window.chrome) return { status: "danger", detail: "window.chrome 完全缺失" };
          return { status: "safe", detail: "window.chrome 存在" };
        },
      },
      {
        name: "chrome.runtime 深度检查",
        label,
        weight: 3,
        fn: () => {
          if (!window.chrome) return { status: "danger", detail: "chrome 对象不存在" };
          const r = window.chrome.runtime;
          if (!r) return { status: "danger", detail: "chrome.runtime 缺失" };
          const props = ["connect", "sendMessage", "getManifest", "id", "getURL", "onMessage", "onConnect"];
          const missing = props.filter((p) => r[p] === undefined);
          if (missing.length > 0) return { status: "danger", detail: "chrome.runtime 缺少属性: " + missing.join(", ") };
          return { status: "safe", detail: "chrome.runtime 完整 (" + props.length + "/7 属性正常)" };
        },
      },
      {
        name: "chrome.loadTimes 存在性",
        label,
        weight: 1,
        fn: () => {
          if (typeof window.chrome?.loadTimes === "function") {
            return { status: "safe", detail: "chrome.loadTimes 存在（Chrome 特有，headless 常缺失）" };
          }
          return { status: "warning", detail: "chrome.loadTimes 缺失" };
        },
      },
      {
        name: "chrome.csi 存在性",
        label,
        weight: 1,
        fn: () => {
          if (typeof window.chrome?.csi === "function") {
            return { status: "safe", detail: "chrome.csi 存在" };
          }
          return { status: "warning", detail: "chrome.csi 缺失" };
        },
      },
      {
        name: "chrome.app 是否存在",
        label,
        weight: 1,
        fn: () => {
          if (window.chrome && window.chrome.app) {
            return { status: "warning", detail: "chrome.app 存在（过时 API，部分自动化工具会漏掉）" };
          }
          return { status: "safe", detail: "chrome.app 不存在" };
        },
      },
    ];
  }

  groupConsole() {

    return [
      {
        name: "console.debug 是否被篡改",
        label,
        weight: 3,
        fn: () => {
          try {
            const str = Function.prototype.toString.call(console.debug);
            if (!str.includes("native code")) {
              return { status: "danger", detail: "console.debug 的 toString 不包含 native code，已被劫持" };
            }
            return { status: "safe", detail: "console.debug 为原生函数" };
          } catch (e) {
            return { status: "danger", detail: "console.debug.toString 调用异常: " + e.message };
          }
        },
      },
      {
        name: "console.log 是否被篡改",
        label,
        weight: 2,
        fn: () => {
          try {
            const str = Function.prototype.toString.call(console.log);
            if (!str.includes("native code")) {
              return { status: "danger", detail: "console.log 已被劫持" };
            }
            return { status: "safe", detail: "console.log 为原生函数" };
          } catch (e) {
            return { status: "danger", detail: "调用异常: " + e.message };
          }
        },
      },
      {
        name: "console 对象自身是否被冻结",
        label,
        weight: 2,
        fn: () => {
          try {
            const saved = console.log;
            console.log = 1;
            if (typeof console.log === "number") {
              console.log = saved;
              return { status: "safe", detail: "console 属性可写（正常浏览器行为）" };
            }
            return { status: "safe", detail: "console 正常" };
          } catch (_) {
            return { status: "danger", detail: "console 属性被冻结 (Object.freeze)，可能是 anti-cdp 手段" };
          }
        },
      },
      {
        name: "CDP 堆栈探测",
        label,

          try {
            const err = new Error("CDP_PROBE");

        label,
        weight: 1,

            if (!str.includes("native code")) return { status: "danger", detail: "console.trace 被篡改" };

            return { status: "warning", detail: "console.trace 检测异常" };

      },
    ];
  }
