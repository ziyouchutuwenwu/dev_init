```elixir
defmodule DemoZig do
  @target System.get_env("ZIG_TARGET")
  @mod System.get_env("ZIG_MOD")

  flags = if(@target, do: ["-Dtarget=#{@target}"], else: [])
  flags = if(@mod, do: flags ++ ["-Doptimize=#{@mod}"], else: flags)

  import Zig, only: [sigil_Z: 2]

  prebuilt = Path.wildcard("priv/native/Elixir.*.so")

  if prebuilt != [] do
    [so_file | _] = prebuilt
    IO.puts("precompiled: #{so_file}")
    use Zig, otp_app: :demo, precompiled: so_file
  else
    tag = if(@target, do: " target=#{@target}")
    IO.puts("[zigler] build from source#{tag}")

    use Zig,
      otp_app: :demo,
      build_flags: flags,
      extra_modules: [
        libroot: {"./native/zig_demo/src/lib_root.zig", []}
      ]
  end

  ......
end
```

lib/mix/tasks/zigler_build.ex

```elixir
defmodule Mix.Tasks.Zigler.Build do
  use Mix.Task

  @shortdoc "MIX_ENV=prod mix zigler.build --target x86_64-linux-gnu.2.17"

  def run(args) do
    {opts, _} = OptionParser.parse!(args, strict: [target: :string])

    target = opts[:target] || ""
    optimize = System.get_env("ZIG_MOD") || "ReleaseSafe"

    env =
      [
        {"ZIG_TARGET", target},
        {"ZIG_MOD", optimize}
      ]
      |> Enum.into(System.get_env())

    File.mkdir_p!("priv/native")

    System.cmd("mix", ["compile", "--force"],
      cd: File.cwd!(),
      env: env,
      into: IO.stream(:stdio, :line),
      stderr_to_stdout: true
    )

    mix_env = System.get_env("MIX_ENV") || "dev"
    app = Mix.Project.config()[:app]
    dir = "_build/#{mix_env}/lib/#{app}/priv/lib"

    dir
    |> File.ls!()
    |> Enum.filter(&String.starts_with?(&1, "Elixir."))
    |> Enum.filter(&String.ends_with?(&1, ".so"))
    |> case do
      [] -> Mix.raise("no Elixir.*.so found in #{dir}")
      files -> Enum.each(files, &File.cp!(Path.join(dir, &1), "priv/native/#{&1}"))
    end

    Mix.shell().info("ok: priv/native/")
  end
end
```
