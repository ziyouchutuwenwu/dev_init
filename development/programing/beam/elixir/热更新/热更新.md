# 热更新

## 说明

每个 beam 会在内存里面同时保留两个版本

通过 MFA 形式的方法调用，会被更新

如果 MFA 调用的方法内还调用了其它方法，不管什么形式调用，被调用的模块都会被更新

## 例子

```elixir
defmodule Demo do
  require Logger

  def start do
    spawn(__MODULE__, :run, [])
  end

  def run do
    demo1()
    __MODULE__.demo2()
    Process.sleep(5_000)
    run()
  end

  def demo1 do
    sub_demo1()
    Logger.debug("demo1 old")
  end

  def sub_demo1 do
    Logger.debug("sub_demo1 old")
  end

  def demo2 do
    sub_demo2()
    Logger.debug("demo2 old")
  end

  def sub_demo2 do
    Logger.debug("sub_demo2 old")
  end
end
```

## 手动控制

```elixir
defmodule Upgrading do
  defmacro __using__(opts) do
    from_version = Keyword.get(opts, :from, 1.0)
    to_version = Keyword.get(opts, :to, 1.0)

    quote do
      require Logger

      @from_version unquote(from_version)
      @to_version unquote(to_version)

      def start_process(args) do
        spawn(__MODULE__, :loop, [args, @from_version])
      end

      def show_version(version) do
        Logger.debug("current version #{inspect(version)}")
      end

      def do_upgrade() do
        Logger.debug("code upgrading to version #{inspect(@to_version)}")
        @to_version
      end

      def loop(args, version) do
        receive do
          :upgrade ->
            new_version = __MODULE__.do_upgrade()
            __MODULE__.loop(args, new_version)

          :version ->
            __MODULE__.show_version(version)
            __MODULE__.loop(args, version)

          _ ->
            __MODULE__.loop(args, version)
        end
      end
    end
  end
end
```

调用

```elixir
defmodule Demo do
  require Logger

  use Upgrading, from: 1.1, to: 2.0

  def start do
    pid = start_process([])
    Process.register(pid, :aaa)
  end

  def version() do
    send(:aaa, :version)
  end

  def upgrade() do
    send(:aaa, :upgrade)
  end
end
```
