# 尾递归

## 说明

最后一行调用自身，不会爆栈

## 例子

向下递减

```erlang
-module(aa).

-compile(export_all).

aaa(N) ->
  aaa(N -1, N *10).

aaa(Index, Val) when Index > 0->
  aaa(Index - 1, Val + Index * 10);

aaa(0, Value) ->
  Value.
```

向上递增

```erlang
-module(bb).

-compile(export_all).

aaa(N) ->
  aaa(0, N *10, N).

aaa(LoopIndex, Val,  EndIndex) when LoopIndex < EndIndex ->
  aaa(LoopIndex + 1, Val + LoopIndex * 10, EndIndex);

aaa(LoopIndex, Value, EndIndex) when LoopIndex == EndIndex  ->
  Value.
```
