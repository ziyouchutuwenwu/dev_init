# 多组件共享

## 说明

多组件共享，非全局

## 例子

### 准备工作

```sh
ng g c demo
ng g c demo/demo1
ng g c demo/demo2
ng g s service/demo-service
```

### service

demo-service.ts

```typescript
import { Injectable } from "@angular/core";

// 这里不全局注入
@Injectable()
export class DemoService {
  private _value = -1;

  constructor() {
    console.log("on service constructor");
    this._value = Math.random();
  }

  get() {
    return this._value;
  }
}
```

### 组件

demo.html

```html
<router-outlet></router-outlet>
```

demo.ts

```typescript
import { Component } from "@angular/core";
import { RouterOutlet } from "@angular/router";
import { DemoService } from "../service/demo-service";

@Component({
  selector: "app-demo",
  imports: [RouterOutlet],
  templateUrl: "./demo.html",
  styleUrl: "./demo.css",
  // 此时作为父组件，子组件共享 DemoService
  // providers: [DemoService],
})
export class Demo {}
```

demo1.ts

```typescript
import { Component } from "@angular/core";
import { DemoService } from "../../service/demo-service";

@Component({
  selector: "app-demo1",
  imports: [],
  templateUrl: "./demo1.html",
  styleUrl: "./demo1.css",
})
export class Demo1 {
  constructor(private demoService: DemoService) {}

  ngOnInit(): void {
    console.log("demo1 get", this.demoService.get());
  }
}
```

### 注册方式 1

app.routes.ts

```typescript
import { Routes } from "@angular/router";
import { Demo } from "./demo/demo";
import { DemoService } from "./service/demo-service";
import { Demo1 } from "./demo/demo1/demo1";
import { Demo2 } from "./demo/demo2/demo2";

export const routes: Routes = [
  { path: "", redirectTo: "demo", pathMatch: "full" },
  {
    path: "demo",
    component: Demo,
    // 通过路由注册
    providers: [DemoService],
    children: [
      { path: "", redirectTo: "demo1", pathMatch: "full" },
      { path: "demo1", component: Demo1 },
      { path: "demo2", component: Demo2 },
    ],
  },
];
```

### 注册方式 2

app.routes.ts

```typescript
import { Routes } from "@angular/router";
import { Demo } from "./demo/demo";
import { Demo1 } from "./demo/demo1/demo1";
import { Demo2 } from "./demo/demo2/demo2";

export const routes: Routes = [
  { path: "", redirectTo: "demo", pathMatch: "full" },
  {
    path: "demo",
    component: Demo,
    // 通过父组件注册,demo.ts 里面，启用 providers
    children: [
      { path: "", redirectTo: "demo1", pathMatch: "full" },
      { path: "demo1", component: Demo1 },
      { path: "demo2", component: Demo2 },
    ],
  },
];
```
