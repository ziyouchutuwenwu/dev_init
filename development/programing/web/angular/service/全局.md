### service

demo-service.ts

```typescript
import { Injectable } from "@angular/core";

@Injectable({
  providedIn: "root",
})
export class DemoService {
  private _value = -1;

  constructor() {
    console.log("on service constructor");
  }

  get() {
    console.log("on service get", this._value);
    return this._value;
  }

  set(value: number) {
    console.log("on service set");
    this._value = value;
  }
}
```

### 组件

demo1.ts

```typescript
import { Component, OnInit } from "@angular/core";
import { DemoService } from "../service/demo-service";

@Component({
  selector: "app-demo1",
  imports: [],
  templateUrl: "./demo1.html",
  styleUrl: "./demo1.css",
})
export class Demo1 implements OnInit {
  constructor(private demoService: DemoService) {
    console.log("app-demo1");
  }

  ngOnInit(): void {
    this.demoService.set(11111);
  }
}
```

demo2.ts

```typescript
import { Component, OnInit } from "@angular/core";
import { DemoService } from "../service/demo-service";

@Component({
  selector: "app-demo2",
  imports: [],
  templateUrl: "./demo2.html",
  styleUrl: "./demo2.css",
})
export class Demo2 implements OnInit {
  public showValue = 0;
  constructor(private demoService: DemoService) {
    console.log("app-demo2");
  }

  ngOnInit(): void {
    this.showValue = this.demoService.get();
  }
}
```

app.routes.ts

```typescript

import { Component } from "@angular/core";
```
