# 路由参数

## 说明

```sh
/aaa?xx=yy

# 动态参数不支持 ssr
/bbb/:xx
```

## 例子

### 创建项目

```sh
ng new demo
cd demo
ng g c aaa
ng g c bbb
```

### 路由

```sh
app.routes.ts
```

```typescript
import { Routes } from "@angular/router";
import { AaaComponent } from "./aaa/aaa.component";
import { BbbComponent } from "./bbb/bbb.component";

export const routes: Routes = [
  {
    path: "aaa",
    component: AaaComponent,
  },
  {
    path: "bbb/:id",
    component: BbbComponent,
  },
];
```

### 问号参数

aaa.html

```html
<p>问号作为参数 id 是 {{id}}</p>
```

aaa.ts

```typescript
import { Component, OnInit } from "@angular/core";
import { ActivatedRoute } from "@angular/router";

@Component({
  selector: "app-aaa",
  imports: [],
  templateUrl: "./aaa.html",
  styleUrl: "./aaa.css",
})
export class Aaa implements OnInit {
  public id: number;

  constructor(private router: ActivatedRoute) {
    this.id = 0;
  }

  ngOnInit() {
    this.router.queryParams.subscribe((params) => {
      this.id = params["id"];
    });
  }
}
```

### uri 路径参数

bbb.html

```html
<p>uri 路径作为参数 id 是 {{id}}</p>
```

bbb.ts

```typescript
import { Component, OnInit } from "@angular/core";
import { ActivatedRoute } from "@angular/router";

@Component({
  selector: "app-bbb",
  imports: [],
  templateUrl: "./bbb.html",
  styleUrl: "./bbb.css",
})
export class Bbb implements OnInit {
  public id: number;

  constructor(private router: ActivatedRoute) {
    this.id = 0;
  }

  ngOnInit() {
    this.router.params.subscribe((param) => {
      this.id = param["id"];
    });
  }
}
```

### 调用处

app.html

```html
<div>
  <a [routerLink]="['/aaa']" [queryParams]="{ id: 100 }">问号参数静态调用</a>
  <input type="button" value="问号参数动态调用" (click)="demo1()" />
</div>
<div>
  <a [routerLink]="['/bbb', 1]">url 参数静态调用</a>
  <input type="button" value="uri 路径参数动态调用" (click)="demo2()" />
</div>
<router-outlet></router-outlet>
```

app.ts

```typescript
import { Component } from "@angular/core";
import { Router, RouterOutlet, RouterLink } from "@angular/router";

@Component({
  selector: "app-root",
  imports: [RouterOutlet, RouterLink],
  templateUrl: "./app.html",
  styleUrl: "./app.css",
})
export class App {
  protected title = "demo";

  constructor(private router: Router) {}

  demo1() {
    this.router.navigate(["/aaa"], {
      queryParams: {
        id: 101,
      },
      // 可以不写，默认为false，设为true时路由跳转浏览器中的url会保持不变，传入的参数依然有效
      skipLocationChange: true,
    });
  }

  demo2() {
    this.router.navigate(["/bbb", 20]);
  }
}
```
